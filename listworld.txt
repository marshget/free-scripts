-- ambil nama bot
local botName = getBot().name

-- tentukan path desktop
-- Windows (ganti USERNAME dengan nama user di PC kamu)
local desktopPath = "C:/Users/USERNAME/Desktop/botname.txt"

-- Linux/Android (biasanya folder Home/Desktop)
-- local desktopPath = os.getenv("HOME") .. "/Desktop/botname.txt"

-- buka file untuk menulis
local file = io.open(desktopPath, "w")
if file then
    file:write(botName)
    file:close()
    print("Nama bot berhasil disimpan di Desktop: " .. botName)
else
    print("Gagal membuat file di Desktop.")
end
